Name badge — uniform scale

// 1. Render both rows at base size (no wrap)
// 2. Measure unscaled dimensions
// 3. Scale uniformly to fit padded area
const PAD = 8, AW = W - PAD*2, AH = H - PAD*2;

function fitName(inner) {
  inner.style.transform = 'scale(1)';
  requestAnimationFrame(() => {
    const s = Math.min(AW / inner.offsetWidth,
                       AH / inner.offsetHeight, 1);
    inner.style.transform = `scale(${s})`;
  });
}

Text badge — shrink font to ≤ 3 rows, then scale

// 1. Binary-search font size until text fits in ≤ 3 rows
// 2. Safety scale pass to guarantee no overflow
const LINE_H = 1.2, MAX_ROWS = 3;

function fitText(inner, el) {
  let lo = 4, hi = 20, best = 4;
  inner.style.transform = 'scale(1)';
  for (let i = 0; i < 16; i++) {
    const mid = (lo + hi) / 2;
    el.style.fontSize = mid + 'px';
    if (inner.offsetHeight <= mid * LINE_H * MAX_ROWS + 0.5)
      { best = mid; lo = mid; } else hi = mid;
  }
  el.style.fontSize = best + 'px';
  requestAnimationFrame(() => {
    const s = Math.min(AW / inner.offsetWidth,
                       AH / inner.offsetHeight, 1);
    inner.style.transform = `scale(${s})`;
  });
}
